    def __init__(self, val: int):

class MyLinkedList:
    def __init__(self):
        self.head = ListNode(0)
        self.size = 0

    def get(self, index: int) -> int:

        cur = self.head.next

        return cur.val

    ListNode tail;
    int size;

        tail = new ListNode(0);
        head.next = tail;
        tail.prev = head;

        if (index <= size / 2) {
            ListNode cur = head;
            for (int i = 0; i < index; i++) {
                cur = cur.next;
            }
            return cur;
        } else {
            ListNode cur = tail;
            for (int i = 0; i < size - index + 1; i++) {
                cur = cur.prev;
            }
            return cur;
        }
    }
