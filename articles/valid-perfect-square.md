## 1. Brute Force

::tabs-start

```python
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        for i in range(1, num + 1):
            sq = i * i
            if sq > num:
                return False
            if sq == num:
                return True
```

```java
public class Solution {
    public boolean isPerfectSquare(int num) {
        for (long i = 1; i <= num; i++) {
            long sq = i * i;
            if (sq > num) {
                return false;
            }
            if (sq == num) {
                return true;
            }
        }
        return false;
    }
}
```

## 3. Binary Search

::tabs-start

```python
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        l, r = 1, num

        while l <= r:
            m = l + (r - l) // 2
            sq = m * m
            if sq > num:
                r = m - 1
            elif sq < num:
                l = m + 1
            else:
                return True

        return False
```

```java
public class Solution {
    public boolean isPerfectSquare(int num) {

            if (sq > num) {
                r = m - 1;
            } else if (sq < num) {
                l = m + 1;
            } else {
                return true;
            }
        }

        return false;
    }
}
```

## 5. Newton's Method

::tabs-start

```python
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        r = num
        while r * r > num:
            r = (r + (num // r)) // 2
        return r * r == num
```

```java
public class Solution {
    public boolean isPerfectSquare(int num) {
        long r = num;
        while (r * r > num) {
            r = (r + num / r) / 2;
        }
        return r * r == num;
    }
}
```

```cpp
class Solution {
public:
    bool isPerfectSquare(int num) {
        long long r = num;
        while (r * r > num) {
            r = (r + num / r) / 2;
        }
        return r * r == num;
    }
};
```

```javascript
class Solution {
    /**
     * @param {number} num
     * @return {boolean}
     */
    isPerfectSquare(num) {
        let r = num;
        while (r * r > num) {
            r = Math.floor((r + Math.floor(num / r)) / 2);
        }
        return r * r === num;
```

## 6. Bit Manipulation

::tabs-start

```python
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        r, mask = 0, 1 << 15

        while mask > 0:
            r |= mask
            if r > (num // r):
                r ^= mask
            mask >>= 1
        
        return r * r == num
```

```java
public class Solution {
    public boolean isPerfectSquare(int num) {
        int r = 0, mask = 1 << 15;

        while (mask > 0) {
            r |= mask;
            if (r > (num / r)) {
                r ^= mask;
            }
            mask >>= 1;
        }
        
        return r * r == num;
    }
}
```
